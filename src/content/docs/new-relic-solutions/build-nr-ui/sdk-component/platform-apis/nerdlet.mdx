A set of methods to modify the nerdlet state.

Use `NerdletStateContext` to read the nerdlet state.

### Usage

```js
import { nerdlet } from 'nr1'
```

### Methods

### `nerdlet.getSetUrlStateLocation`

<FunctionDefinition
  arguments={[{"defaultValue":null,"description":"New nerdlet URL state.","name":"urlState","type":"Object"}]}
  returnValue={{"description": null, "type": "undefined"}}
/>

```js
nerdlet.getSetUrlStateLocation({
  foo: 'bar',
});
```

### `nerdlet.setConfig`

<FunctionDefinition
  arguments={[{"defaultValue":null,"description":"New nerdlet config","name":"config","type":"NerdletConfig"}]}
  returnValue={{"description": null, "type": "undefined"}}
/>

```js
nerdlet.setConfig({
  accountPicker: true,
});
```

```js
nerdlet.setConfig({
  accountPicker: true,
  accountPickerValues: [
    nerdlet.ACCOUNT_PICKER_VALUE.CROSS_ACCOUNT,
    ...nerdlet.ACCOUNT_PICKER_DEFAULT_VALUES,
  ],
});
```

```js
nerdlet.setConfig({
  actionControls: true,
  actionControlButtons: [
    {
      label: 'Favorite',
      type: nerdlet.ACTION_CONTROL_BUTTON_TYPES.SECONDARY,
      hint: 'Add entity to my favorites',
      iconType: Icon.TYPE.PROFILES__EVENTS__FAVORITE,
      onClick: () => this.onFavoriteEntity(),
    },
  ],
});
```

```js
nerdlet.setConfig({
  filterBar: true,
});
```

```js
nerdlet.setConfig({
  timePicker: true,
  timePickerRanges: nerdlet.TIME_PICKER_DEFAULT_RANGES,
});
```

```js
nerdlet.setConfig({
  timePicker: true,
  timePickerRanges: [
    nerdlet.TIME_PICKER_RANGE.NONE,
    {
      label: 'Last 15 days',
      offset: 1000 * 60 * 60 * 24 * 15,
    },
    nerdlet.TIME_PICKER_RANGE.CUSTOM,
  ],
});
```

```js
nerdlet.setConfig({
  header: false,
});
```

```js
nerdlet.setConfig({
  headerType: nerdlet.HEADER_TYPE.CUSTOM,
  headerTitle: 'My custom header title',
  headerTitleMetadata: 'My custom header title metadata',
});
```

```js
nerdlet.setConfig({
  headerType: nerdlet.HEADER_TYPE.CUSTOM,
  headerTitle: 'My custom header title',
  headerParentTitle: 'Parent nerdlet I want to redirect to',
  headerParentLocation: navigation.openNerdlet({
    id: 'my-nerdlet-to-redirect-to',
  }),
});
```
