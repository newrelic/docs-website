## Examples

### nrEnableHttpEvents

```js
  	nrEnableHttpEvents(nr)
```

### nrDisableHttpEvents

```js
  	nrDisableHttpEvents(nr)
```

### nrSendHttpRequest

```js
  	urlReq = CreateObject("roUrlTransfer")
	  urlReq.SetUrl(_url)
	  ...
	  nrSendHttpRequest(m.nr, urlReq)
```

### nrSendHttpResponse

```js
  	msg = wait(5000, m.port)
	  if type(msg) = "roUrlEvent" then
		        nrSendHttpResponse(m.nr, _url, msg)
	  end if
```
