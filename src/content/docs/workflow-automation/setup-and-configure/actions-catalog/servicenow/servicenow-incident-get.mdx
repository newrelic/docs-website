Gets an existing incident from ServiceNow using the Table API.

For authentication setup, see [ServiceNow Incident actions](/docs/workflow-automation/setup-and-configure/actions-catalog/servicenow/servicenow-incident).

  <TabsPageItem id="servicenow-incident-get-example">
  <table>
    <thead>
        <tr>
            <th>Workflow example</th>
        </tr>
    </thead>
    <tbody>
        <tr>
            <td>
```yaml
name: get-servicenow-incident
description: to get a incident from ServiceNow
steps:
  - type: action
    name: getIncident
    action: serviceNow.incident.get
    version: 1
    inputs:
      domain: "your-domain"
      apiKey: "${{ :secrets:serviceNowApiKey }}"
      sysId: "ed59ab193b67ba10e811026aa5e45a2d"
      parameters:
        sysparm_fields: "number,short_description,description,priority"
      selectors:
        - name: sysId
          expression: .responseBody.result.sys_id
    next: end
```
            </td>
