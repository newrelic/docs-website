            </th>

        <tbody>
          <tr>
            <td>
              ```yaml
              name: get-servicenow-incident-list
              description: to get list of incident from ServiceNow
              steps:
                - type: action
                  name: listIncidents
                  action: serviceNow.incident.list
                  version: 1
                  inputs:
                    domain: "your-domain"
                    apiKey: "${{ :secrets:serviceNowApiKey }}"
                    parameters:
                      sysparm_fields: "number,short_description,description"
                      sysparm_limit: "5"
                    selectors:
                      - name: incidents
                        expression: .responseBody.result
                  next: end
              ```
