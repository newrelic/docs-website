    <TabsPageItem id="runagentexample">
      ```yaml
      name: demo-run-agent
      description: Run newrelic word problem agent
      steps:
        - name: runAgent
          type: action
          action: newrelic.agent.run
          version: '1'
          inputs:
            prompt: Sum of prime number less than 100
            agentId: word_problem_agent
            context:
              - description: accountId
                value: "12345678"
              - description: slack_channel_id
                value: "CXXXXXXXXX"
            selectors:
              - name: success
                expression: .success
              - name: error
                expression: .errorMessage
              - name: response
                expression: .response
      ```
